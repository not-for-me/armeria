17  package com.linecorp.armeria.server.docs;
18  
19  import org.apache.thrift.protocol.TType;
20  
21  enum ValueType {
22      STOP,
23      VOID,
24      BOOL,
25      BYTE,
26      DOUBLE,
27      I16,
28      I32,
29      I64,
30      STRING,
31      STRUCT,
32      MAP,
33      SET,
34      LIST,
35      ENUM;
36  
37      static ValueType of(byte type) {
38          switch (type) {
39          case TType.STOP:
40              return ValueType.STOP;
41          case TType.VOID:
42              return ValueType.VOID;
43          case TType.BOOL:
44              return ValueType.BOOL;
45          case TType.BYTE:
46              return ValueType.BYTE;
47          case TType.DOUBLE:
48              return ValueType.DOUBLE;
49          case TType.I16:
50              return ValueType.I16;
51          case TType.I32:
52              return ValueType.I32;
53          case TType.I64:
54              return ValueType.I64;
55          case TType.STRING:
56              return ValueType.STRING;
57          case TType.STRUCT:
58              return ValueType.STRUCT;
59          case TType.MAP:
60              return ValueType.MAP;
61          case TType.SET:
62              return ValueType.SET;
63          case TType.LIST:
64              return ValueType.LIST;
65          case TType.ENUM:
66              return ValueType.ENUM;
67          default:
68              throw new IllegalArgumentException("unknown field value type: " + type);
69          }
70      }
71  }

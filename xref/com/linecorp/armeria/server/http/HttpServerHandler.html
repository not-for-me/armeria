81      private static final Set<HttpMethod> ALLOWED_METHODS =
82              Sets.immutableEnumSet(HttpMethod.DELETE, HttpMethod.GET, HttpMethod.HEAD, HttpMethod.OPTIONS,
83                                    HttpMethod.PATCH, HttpMethod.POST, HttpMethod.PUT, HttpMethod.TRACE);
84  
85      private static final String ALLOWED_METHODS_STRING =
86              ALLOWED_METHODS.stream().map(HttpMethod::name).collect(Collectors.joining(","));
87  
88      private static final Pattern PROHIBITED_PATH_PATTERN =
89              Pattern.compile("(?:[:<>|?*\\\\]|/\\.\\.|\\.\\.$|\\.\\./|//+)");

91      private static final ChannelFutureListener CLOSE = future -> {
92          final Throwable cause = future.cause();
93          final Channel ch = future.channel();
94          if (cause != null) {
95              Exceptions.logIfUnexpected(logger, ch, HttpServer.get(ch).protocol(), cause);
96          }
97          safeClose(ch);
98      };
99  
100     static final ChannelFutureListener CLOSE_ON_FAILURE = future -> {
101         final Throwable cause = future.cause();
102         if (cause != null && !(cause instanceof ClosedPublisherException)) {
103             final Channel ch = future.channel();
104             Exceptions.logIfUnexpected(logger, ch, HttpServer.get(ch).protocol(), cause);
105             safeClose(ch);
106         }
107     };
108 
109     static void safeClose(Channel ch) {
110         if (!ch.isActive()) {
111             return;
112         }
113 
114         // Do not call Channel.close() if AbstractHttp2ConnectionHandler.close() has been invoked
115         // already. Otherwise, it can trigger a bad cycle:
116         //
117         //   1. Channel.close() triggers AbstractHttp2ConnectionHandler.close().
118         //   2. AbstractHttp2ConnectionHandler.close() triggers Http2Stream.close().
119         //   3. Http2Stream.close() fails the promise of its pending writes.
120         //   4. The failed promise notifies this listener (CLOSE_ON_FAILURE).
121         //   5. This listener calls Channel.close().
122         //   6. Repeat from step 1.
123         //
124 
125         final AbstractHttp2ConnectionHandler h2handler =
126                 ch.pipeline().get(AbstractHttp2ConnectionHandler.class);
127 
128         if (h2handler == null || !h2handler.isClosing()) {
129             ch.close();
130         }
131     }

141     HttpServerHandler(ServerConfig config, SessionProtocol protocol) {
142         assert protocol == SessionProtocol.H1 ||
143                protocol == SessionProtocol.H1C ||
144                protocol == SessionProtocol.H2;
145 
146         this.config = requireNonNull(config, "config");
147         this.protocol = requireNonNull(protocol, "protocol");
148 
149         if (protocol == SessionProtocol.H1 || protocol == SessionProtocol.H1C) {
150             responseEncoder = new Http1ObjectEncoder(true);
151         }
152     }

171     @Override
172     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
173         isReading = true; // Cleared in channelReadComplete()
174 
175         if (msg instanceof Http2Settings) {
176             handleHttp2Settings(ctx, (Http2Settings) msg);
177         } else {
178             handleRequest(ctx, (DecodedHttpRequest) msg);
179         }
180     }
181 
182     private void handleHttp2Settings(ChannelHandlerContext ctx, Http2Settings h2settings) {
183         if (h2settings.isEmpty()) {
184             logger.trace("{} HTTP/2 settings: <empty>", ctx.channel());
185         } else {
186             logger.debug("{} HTTP/2 settings: {}", ctx.channel(), h2settings);
187         }
188 
189         switch (protocol) {
190             case H1:
191                 protocol = SessionProtocol.H2;
192                 break;
193             case H1C:
194                 protocol = SessionProtocol.H2C;
195                 break;
196         }
197 
198         final Http2ConnectionHandler handler = ctx.pipeline().get(Http2ConnectionHandler.class);
199         if (responseEncoder != null) {
200             responseEncoder.close();
201         }
202         responseEncoder = new Http2ObjectEncoder(handler.encoder());
203     }
204 
205     private void handleRequest(ChannelHandlerContext ctx, DecodedHttpRequest req) throws Exception {
206         // Ignore the request received after the last request,
207         // because we are going to close the connection after sending the last response.
208         if (handledLastRequest) {
209             return;
210         }
211 
212         // If we received the message with keep-alive disabled,
213         // we should not accept a request anymore.
214         if (!req.isKeepAlive()) {
215             handledLastRequest = true;
216         }
217 
218         final HttpHeaders headers = req.headers();
219         if (!ALLOWED_METHODS.contains(headers.method())) {
220             respond(ctx, req, HttpStatus.METHOD_NOT_ALLOWED);
221             return;
222         }
223 
224         // Handle 'OPTIONS * HTTP/1.1'.
225         final String originalPath = headers.path();
226         if (originalPath.isEmpty() || originalPath.charAt(0) != '/') {
227             if (headers.method() == HttpMethod.OPTIONS && "*".equals(originalPath)) {
228                 handleOptions(ctx, req);
229             } else {
230                 respond(ctx, req, HttpStatus.BAD_REQUEST);
231             }
232             return;
233         }
234 
235         // Validate and split path and query.
236         final String path = validatePathAndStripQuery(originalPath);
237         if (path == null) {
238             // Reject requests without a valid path.
239             respond(ctx, req, HttpStatus.NOT_FOUND);
240             return;
241         }
242 
243         if (headers.authority() == null) {
244             // HTTP/1.1 requests require a host header.
245             respond(ctx, req, HttpStatus.BAD_REQUEST);
246             return;
247         }
248 
249         final String hostname = hostname(headers);
250         final VirtualHost host = config.findVirtualHost(hostname);
251 
252         // Find the service that matches the path.
253         final PathMapped<ServiceConfig> mapped = host.findServiceConfig(path);
254         if (!mapped.isPresent()) {
255             // No services matched the path.
256             handleNonExistentMapping(ctx, req, host, path);
257             return;
258         }
259 
260         // Decode the request and create a new invocation context from it to perform an invocation.
261         final String mappedPath = mapped.mappedPath();
262         final ServiceConfig serviceCfg = mapped.value();
263         final Service<Request, HttpResponse> service = serviceCfg.service();
264 
265         final Channel channel = ctx.channel();
266         final ServiceRequestContext reqCtx = new DefaultServiceRequestContext(
267                 serviceCfg, channel,
268                 protocol,
269                 req.method().name(), path, mappedPath,
270                 LoggerFactory.getLogger(serviceCfg.loggerName()), req);
271 
272         final RequestLogBuilder reqLogBuilder = reqCtx.requestLogBuilder();
273         final HttpResponse res;
274         try (PushHandle ignored = RequestContext.push(reqCtx)) {
275             req.init(reqCtx);
276             res = service.serve(reqCtx, req);
277         } catch (Throwable cause) {
278             reqLogBuilder.end(cause);
279             if (cause instanceof ResourceNotFoundException) {
280                 respond(ctx, req, HttpStatus.NOT_FOUND);
281             } else if (cause instanceof ServiceUnavailableException) {
282                 respond(ctx, req, HttpStatus.SERVICE_UNAVAILABLE);
283             } else {
284                 logger.warn("{} Unexpected exception: {}, {}", reqCtx, service, req, cause);
285                 respond(ctx, req, HttpStatus.INTERNAL_SERVER_ERROR);
286             }
287             return;
288         }
289 
290         final EventLoop eventLoop = channel.eventLoop();
291 
292         // Keep track of the number of unfinished requests and
293         // clean up the request stream when response stream ends.
294         unfinishedRequests++;
295         res.closeFuture().handle(voidFunction((ret, cause) -> {
296             req.abort();
297             if (cause == null) {
298                 reqLogBuilder.end();
299             } else {
300                 reqLogBuilder.end(cause);
301             }
302             eventLoop.execute(() -> {
303                 if (--unfinishedRequests == 0 && handledLastRequest) {
304                     ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(CLOSE);
305                 }
306             });
307         })).exceptionally(CompletionActions::log);
308 
309         res.subscribe(new HttpResponseSubscriber(ctx, responseEncoder, reqCtx, req), eventLoop);
310     }

312     private void handleOptions(ChannelHandlerContext ctx, DecodedHttpRequest req) {
313         respond(ctx, req,
314                 AggregatedHttpMessage.of(
315                         HttpHeaders.of(HttpStatus.OK)
316                                    .set(HttpHeaderNames.ALLOW, ALLOWED_METHODS_STRING)));
317     }

319     private void handleNonExistentMapping(ChannelHandlerContext ctx, DecodedHttpRequest req,
320                                           VirtualHost host, String path) {
321 
322         if (path.charAt(path.length() - 1) != '/') {
323             // Handle the case where /path doesn't exist but /path/ exists.
324             final String pathWithSlash = path + '/';
325             if (host.findServiceConfig(pathWithSlash).isPresent()) {
326                 final String location;
327                 final String originalPath = req.path();
328                 if (path.length() == originalPath.length()) {
329                     location = pathWithSlash;
330                 } else {
331                     location = pathWithSlash + originalPath.substring(path.length());
332                 }
333                 redirect(ctx, req, location);
334                 return;
335             }
336         }
337 
338         respond(ctx, req, HttpStatus.NOT_FOUND);
339     }

341     private static String hostname(HttpHeaders headers) {
342         final CharSequence hostname = headers.authority();
343 
344         final String hostnameStr = hostname.toString();
345         final int hostnameColonIdx = hostnameStr.lastIndexOf(':');
346         if (hostnameColonIdx < 0) {
347             return hostnameStr;
348         }
349 
350         return hostnameStr.substring(0, hostnameColonIdx);
351     }
352 
353     private static String validatePathAndStripQuery(String path) {
354         // Filter out an empty path or a relative path.
355         if (path.isEmpty() || path.charAt(0) != '/') {
356             return null;
357         }
358 
359         // Strip the query string.
360         final int queryPos = path.indexOf('?');
361         if (queryPos >= 0) {
362             path = path.substring(0, queryPos);
363         }
364 
365         try {
366             path = URLDecoder.decode(path, "UTF-8");
367         } catch (IllegalArgumentException ignored) {
368             // Malformed URL
369             return null;
370         } catch (UnsupportedEncodingException e) {
371             // Should never happen
372             throw new Error(e);
373         }
374 
375         // Reject the prohibited patterns.
376         if (PROHIBITED_PATH_PATTERN.matcher(path).find()) {
377             return null;
378         }
379 
380         return path;
381     }

383     private void redirect(ChannelHandlerContext ctx, DecodedHttpRequest req, String location) {
384         respond(ctx, req,
385                 AggregatedHttpMessage.of(
386                         HttpHeaders.of(HttpStatus.TEMPORARY_REDIRECT)
387                                    .set(HttpHeaderNames.LOCATION, location)));
388     }

391     private void respond(ChannelHandlerContext ctx, DecodedHttpRequest req, HttpStatus status) {
392 
393         if (status.code() < 400) {
394             respond(ctx, req, AggregatedHttpMessage.of(HttpHeaders.of(status)));
395             return;
396         }
397 
398         final HttpData content;
399         if (req.method() == HttpMethod.HEAD || ArmeriaHttpUtil.isContentAlwaysEmpty(status)) {
400             content = HttpData.EMPTY_DATA;
401         } else {
402             content = status.toHttpData();
403         }
404 
405         respond(ctx, req,
406                 AggregatedHttpMessage.of(
407                         HttpHeaders.of(status)
408                                    .set(HttpHeaderNames.CONTENT_TYPE, ERROR_CONTENT_TYPE),
409                         content));
410     }
411 
412     private void respond(ChannelHandlerContext ctx, DecodedHttpRequest req, AggregatedHttpMessage res) {
413         if (!handledLastRequest) {
414             addKeepAliveHeaders(req, res);
415             write(ctx, req, res).addListener(CLOSE_ON_FAILURE);
416         } else {
417             // Note that it is perfectly fine not to set the 'content-length' header to the last response
418             // of an HTTP/1 connection. We set it anyway to work around overly strict HTTP clients that always
419             // require a 'content-length' header for non-chunked responses.
420             setContentLength(req, res);
421             write(ctx, req, res).addListener(CLOSE);
422         }
423 
424         if (!isReading) {
425             ctx.flush();
426         }
427     }
428 
429     private ChannelFuture write(ChannelHandlerContext ctx, DecodedHttpRequest req, AggregatedHttpMessage res) {
430         final boolean trailingHeadersEmpty = res.trailingHeaders().isEmpty();
431         final boolean contentAndTrailingHeadersEmpty = res.content().isEmpty() && trailingHeadersEmpty;
432         ChannelFuture future = responseEncoder.writeHeaders(ctx, req.id(), req.streamId(), res.headers(), contentAndTrailingHeadersEmpty);
433         if (!contentAndTrailingHeadersEmpty) {
434             future = responseEncoder.writeData(ctx, req.id(), req.streamId(), res.content(), trailingHeadersEmpty);
435             if (!trailingHeadersEmpty) {
436                 future = responseEncoder.writeHeaders(ctx, req.id(), req.streamId(), res.trailingHeaders(), true);
437             }
438         }
439         return future;
440     }

442     /**
443      * Sets the keep alive header as per:
444      * - http://www.w3.org/Protocols/HTTP/1.1/draft-ietf-http-v11-spec-01.html#Connection
445      */
446     private static void addKeepAliveHeaders(HttpRequest req, AggregatedHttpMessage res) {
447         res.headers().set(HttpHeaderNames.CONNECTION, "keep-alive");
448         setContentLength(req, res);
449     }
450 
451     /**
452      * Sets the 'content-length' header to the response.
453      */
454     private static void setContentLength(HttpRequest req, AggregatedHttpMessage res) {
455         // http://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.4
456         // prohibits to send message body for below cases.
457         // and in those cases, content should be empty.
458         if (req.method() == HttpMethod.HEAD || ArmeriaHttpUtil.isContentAlwaysEmpty(res.status())) {
459             return;
460         }
461         res.headers().setInt(HttpHeaderNames.CONTENT_LENGTH, res.content().length());
462     }

466         isReading = false;
467         ctx.flush();

475     @Override
476     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {

478         if (ctx.channel().isActive()) {
479             ctx.close();
480         }

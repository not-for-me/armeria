45  /**
46   * Builds a new {@link Server} and its {@link ServerConfig}.
47   * <h2>Example</h2>
48   * <pre>{@code
49   * ServerBuilder sb = new ServerBuilder();
50   * // Add a port to listen
51   * sb.port(8080, SessionProtocol.HTTP);
52   * // Build and add a virtual host.
53   * sb.virtualHost(new VirtualHost("*.foo.com").serviceAt(...).build());
54   * // Add services to the default virtual host.
55   * sb.serviceAt(...);
56   * sb.serviceUnder(...);
57   * // Build a server.
58   * Server s = sb.build();
59   * }</pre>
60   *
61   * @see VirtualHostBuilder
62   */
63  public final class ServerBuilder {
64  
65      private static final int DEFAULT_NUM_WORKERS;
66      private static final int DEFAULT_MAX_PENDING_REQUESTS = 8;
67      private static final int DEFAULT_MAX_CONNECTIONS = 65536;
68      // Use slightly greater value than the client default so that clients close the connection more often.
69      private static final long DEFAULT_IDLE_TIMEOUT_MILLIS = Duration.ofSeconds(15).toMillis();
70      private static final long DEFAULT_DEFAULT_REQUEST_TIMEOUT_MILLIS = Duration.ofSeconds(10).toMillis();
71      private static final long DEFAULT_DEFAULT_MAX_REQUEST_LENGTH = 10 * 1024 * 1024; // 10 MB
72      // Defaults to no graceful shutdown.
73      private static final Duration DEFAULT_GRACEFUL_SHUTDOWN_QUIET_PERIOD = Duration.ZERO;
74      private static final Duration DEFAULT_GRACEFUL_SHUTDOWN_TIMEOUT = Duration.ZERO;
75      private static final int DEFAULT_MAX_BLOCKING_TASK_THREADS = 200; // from Tomcat's maxThreads.
76      private static final String DEFAULT_SERVICE_LOGGER_PREFIX = "armeria.services";
77  
78      static {
79          String value = System.getProperty("io.netty.eventLoopThreads", "0");
80          final int fallbackDefaultNumWorkers = Runtime.getRuntime().availableProcessors() * 2;
81          int defaultNumWorkers;
82          try {
83              defaultNumWorkers = Integer.parseInt(value);
84              if (defaultNumWorkers <= 0) {
85                  defaultNumWorkers = fallbackDefaultNumWorkers;
86              }
87          } catch (Exception ignored) {
88              defaultNumWorkers = fallbackDefaultNumWorkers;
89          }
90  
91          DEFAULT_NUM_WORKERS = defaultNumWorkers;
92      }

95          return DefaultBlockingTaskExecutorHolder.INSTANCE;

100                 0, DEFAULT_MAX_BLOCKING_TASK_THREADS,
101                 60, TimeUnit.SECONDS, new LinkedTransferQueue<>(),
102                 new DefaultThreadFactory("armeria-blocking-tasks", true));
103     }

242     /**
243      * Sets the amount of time to wait after calling {@link Server#stop()} for
244      * requests to go away before actually shutting down.
245      *
246      * @param quietPeriodMillis the number of milliseconds to wait for active
247      *     requests to go end before shutting down. 0 means the server will
248      *     stop right away without waiting.
249      * @param timeoutMillis the number of milliseconds to wait before shutting
250      *     down the server regardless of active requests. This should be set to
251      *     a time greater than {@code quietPeriodMillis} to ensure the server
252      *     shuts down even if there is a stuck request.
253      */
254     public ServerBuilder gracefulShutdownTimeout(long quietPeriodMillis, long timeoutMillis) {
255         return gracefulShutdownTimeout(
256                 Duration.ofMillis(quietPeriodMillis), Duration.ofMillis(timeoutMillis));
257     }
258 
259     /**
260      * Sets the amount of time to wait after calling {@link Server#stop()} for
261      * requests to go away before actually shutting down.
262      *
263      * @param quietPeriod the number of milliseconds to wait for active
264      *     requests to go end before shutting down. {@link Duration#ZERO} means
265      *     the server will stop right away without waiting.
266      * @param timeout the number of milliseconds to wait before shutting
267      *     down the server regardless of active requests. This should be set to
268      *     a time greater than {@code quietPeriod} to ensure the server shuts
269      *     down even if there is a stuck request.
270      */
271     public ServerBuilder gracefulShutdownTimeout(Duration quietPeriod, Duration timeout) {
272         requireNonNull(quietPeriod, "quietPeriod");
273         requireNonNull(timeout, "timeout");
274         gracefulShutdownQuietPeriod = ServerConfig.validateNonNegative(quietPeriod, "quietPeriod");
275         gracefulShutdownTimeout = ServerConfig.validateNonNegative(timeout, "timeout");
276         ServerConfig.validateGreaterThanOrEqual(gracefulShutdownTimeout, "quietPeriod",
277                                                 gracefulShutdownQuietPeriod, "timeout");
278         return this;
279     }

396     private void defaultVirtualHostBuilderUpdated() {
397         updatedDefaultVirtualHostBuilder = true;
398         if (defaultVirtualHost != null) {
399             throw new IllegalStateException("ServerBuilder.defaultVirtualHost() invoked already.");
400         }
401     }

403     /**
404      * Sets the default {@link VirtualHost}, which is used when no other {@link VirtualHost}s match the
405      * host name of a client request. e.g. the {@code "Host"} header in HTTP or host name in TLS SNI extension
406      *
407      * @throws IllegalStateException if other default {@link VirtualHost} builder methods have been invoked
408      * already, including:
409      * <ul>
410      *   <li>{@link #sslContext(SslContext)}</li>
411      *   <li>{@link #serviceAt(String, Service)}</li>
412      *   <li>{@link #serviceUnder(String, Service)}</li>
413      *   <li>{@link #service(PathMapping, Service)}</li>
414      * </ul>
415      *
416      * @see #virtualHost(VirtualHost)
417      */
418     public ServerBuilder defaultVirtualHost(VirtualHost defaultVirtualHost) {
419         requireNonNull(defaultVirtualHost, "defaultVirtualHost");
420         if (updatedDefaultVirtualHostBuilder) {
421             throw new IllegalStateException("invoked other default VirtualHost builder methods already");
422         }
423 
424         this.defaultVirtualHost = defaultVirtualHost;
425         return this;
426     }

428     public <T extends Service<T_I, T_O>, T_I extends Request, T_O extends Response,
429             R extends Service<R_I, R_O>, R_I extends Request, R_O extends Response> ServerBuilder decorator(
430             Function<T, R> decorator) {
431 
432         requireNonNull(decorator, "decorator");
433 
434         @SuppressWarnings("unchecked")
435         Function<Service<Request, Response>, Service<Request, Response>> castDecorator =
436                 (Function<Service<Request, Response>, Service<Request, Response>>) decorator;
437 
438         if (this.decorator != null) {
439             this.decorator = this.decorator.andThen(castDecorator);
440         } else {
441             this.decorator = castDecorator;
442         }
443 
444         return this;
445     }

447     /**
448      * Creates a new {@link Server} with the configuration properties set so far.
449      */
450     public Server build() {
451         Executor blockingTaskExecutor = this.blockingTaskExecutor;
452         if (blockingTaskExecutor == null) {
453             blockingTaskExecutor = defaultBlockingTaskExecutor();
454         }
455 
456         final List<ServerPort> ports =
457                 !this.ports.isEmpty() ? this.ports
458                                       : Collections.singletonList(new ServerPort(0, SessionProtocol.HTTP));
459 
460         final VirtualHost defaultVirtualHost;
461         if (this.defaultVirtualHost != null) {
462             defaultVirtualHost = this.defaultVirtualHost.decorate(decorator);
463         } else {
464             defaultVirtualHost = defaultVirtualHostBuilder.build().decorate(decorator);
465         }
466 
467         final List<VirtualHost> virtualHosts;
468         if (decorator != null) {
469             virtualHosts = this.virtualHosts.stream()
470                                             .map(h -> h.decorate(decorator))
471                                             .collect(Collectors.toList());
472         } else {
473             virtualHosts = this.virtualHosts;
474         }
475 
476         return new Server(new ServerConfig(
477                 ports, defaultVirtualHost, virtualHosts, numWorkers, maxPendingRequests, maxConnections,
478                 idleTimeoutMillis, defaultRequestTimeoutMillis, defaultMaxRequestLength,
479                 gracefulShutdownQuietPeriod, gracefulShutdownTimeout,
480                 blockingTaskExecutor, serviceLoggerPrefix));
481     }

483     @Override
484     public String toString() {
485         return ServerConfig.toString(
486                 getClass(), ports, defaultVirtualHost, virtualHosts,
487                 numWorkers, maxPendingRequests, maxConnections, idleTimeoutMillis,
488                 defaultRequestTimeoutMillis, defaultMaxRequestLength,
489                 gracefulShutdownQuietPeriod, gracefulShutdownTimeout,
490                 blockingTaskExecutor, serviceLoggerPrefix);
491     }

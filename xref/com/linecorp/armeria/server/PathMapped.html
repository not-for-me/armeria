17  package com.linecorp.armeria.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import javax.annotation.Nullable;
22  
23  /**
24   * A value mapped by {@link PathMappings}.
25   */
26  public final class PathMapped<T> {
27  
28      private static final PathMapped<Object> EMPTY = new PathMapped<>(null, null);
29  
30      /**
31       * Returns a singleton instance of a {@link PathMapped} that represents a non-existent value.
32       */
33      @SuppressWarnings("unchecked")
34      public static <T> PathMapped<T> empty() {
35          return (PathMapped<T>) EMPTY;
36      }
37  
38      /**
39       * Creates a new {@link PathMapped} with the specified {@code mappedPath} and {@code value}.
40       *
41       * @param mappedPath the path translated by {@link PathMappings}
42       * @param value      the value
43       */
44      public static <T> PathMapped<T> of(String mappedPath, T value) {
45          return new PathMapped<>(requireNonNull(mappedPath, "mappedPath"),
46                                  requireNonNull(value, "value"));
47      }
48  
49      @Nullable
50      private final String mappedPath;
51      @Nullable
52      private final T value;
53  
54      private PathMapped(@Nullable String mappedPath, @Nullable T value) {
55          assert mappedPath != null && value != null ||
56                 mappedPath == null && value == null;
57  
58          this.mappedPath = mappedPath;
59          this.value = value;
60      }
61  
62      /**
63       * Returns {@code true} if and only if {@link PathMappings} found a matching value.
64       */
65      public boolean isPresent() {
66          return mappedPath != null;
67      }
68  
69      /**
70       * Returns the path translated by {@link PathMappings}.
71       *
72       * @throws IllegalStateException if there's no match
73       */
74      public String mappedPath() {
75          ensurePresence();
76          return mappedPath;
77      }
78  
79      /**
80       * Returns the value.
81       *
82       * @throws IllegalStateException if there's no match
83       */
84      public T value() {
85          ensurePresence();
86          return value;
87      }
88  
89      private void ensurePresence() {
90          if (!isPresent()) {
91              throw new IllegalStateException("mapping unavailable");
92          }
93      }
94  
95      @Override
96      public String toString() {
97          return isPresent() ? mappedPath() + " -> " + value() : "<empty>";
98      }
99  }

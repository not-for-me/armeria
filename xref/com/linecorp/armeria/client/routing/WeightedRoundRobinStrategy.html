16  package com.linecorp.armeria.client.routing;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.util.HashMap;
21  import java.util.LinkedHashMap;
22  import java.util.List;
23  import java.util.Map;
24  import java.util.concurrent.atomic.AtomicInteger;
25  import java.util.concurrent.atomic.AtomicLong;
26  
27  import com.linecorp.armeria.client.Endpoint;
28  
29  final class WeightedRoundRobinStrategy implements EndpointSelectionStrategy {
30  
31      @Override
32      @SuppressWarnings("unchecked")
33      public EndpointSelector newSelector(EndpointGroup endpointGroup) {
34          return new RoundRobinSelector(endpointGroup);
35      }
36  
37  
38      /**
39       * A weighted round robin select strategy.
40       * <p>
41       * For nodes a,b,c<br>
42       * if endpoint weights are 1,1,1(or 2,2,2),then select result is abc abc ...<br>
43       * if endpoint weights are 1,2,3(or 2,4,6),then select result is abcbcc(or abcabcbcbccc) ...<br>
44       * if endpoint weights are 3,5,7,then select result is abcabcabcbcbcbb abcabcabcbcbcbb ...
45       */
46      static final class RoundRobinSelector implements EndpointSelector {
47          private final EndpointGroup endpointGroup;
48  
49          private final AtomicLong sequence = new AtomicLong();
50  
51          private int minWeight = Integer.MAX_VALUE;
52  
53          private int maxWeight = Integer.MIN_VALUE;
54  
55          private int sumWeight;
56  
57          RoundRobinSelector(EndpointGroup endpointGroup) {
58              requireNonNull(endpointGroup, "endpointGroup");
59  
60              this.endpointGroup = endpointGroup;
61              endpointGroup.endpoints().forEach(e -> {
62                  int weight = e.weight();
63                  minWeight = Math.min(minWeight, weight);
64                  maxWeight = Math.max(maxWeight, weight);
65                  sumWeight += weight;
66              });
67          }
68  
69          @Override
70          public EndpointGroup group() {
71              return endpointGroup;
72          }
73  
74          @Override
75          public EndpointSelectionStrategy strategy() {
76              return EndpointSelectionStrategy.WEIGHTED_ROUND_ROBIN;
77          }
78  
79          @Override
80          public Endpoint select() {
81              List<Endpoint> endpoints = endpointGroup.endpoints();
82              long currentSequence = sequence.getAndIncrement();
83  
84              if (minWeight < maxWeight) {
85                  HashMap<Endpoint, AtomicInteger> endpointWeights = new LinkedHashMap<>();
86                  for (Endpoint endpoint : endpoints) {
87                      endpointWeights.put(endpoint, new AtomicInteger(endpoint.weight()));
88                  }
89  
90                  int mod = (int) (currentSequence % sumWeight);
91                  for (int i = 0; i < maxWeight; i++) {
92                      for (Map.Entry<Endpoint, AtomicInteger> entry : endpointWeights.entrySet()) {
93                          AtomicInteger weight = entry.getValue();
94                          if (mod == 0 && weight.get() > 0) {
95                              return entry.getKey();
96                          }
97                          if (weight.get() > 0) {
98                              weight.decrementAndGet();
99                              mod--;
100                         }
101                     }
102                 }
103             }
104             //endpoints weight equal
105             return endpoints.get((int) (currentSequence % endpoints.size()));
106         }
107     }
108 }

42  import io.netty.bootstrap.Bootstrap;
43  import io.netty.channel.Channel;
44  import io.netty.channel.EventLoop;
45  import io.netty.channel.pool.ChannelHealthChecker;
46  import io.netty.util.concurrent.Future;
47  import io.netty.util.concurrent.FutureListener;
48  
49  final class HttpClientDelegate implements Client<HttpRequest, HttpResponse> {
50  
51      private static final KeyedChannelPoolHandlerAdapter<PoolKey> NOOP_POOL_HANDLER =
52              new KeyedChannelPoolHandlerAdapter<>();
53  
54      private static final ChannelHealthChecker POOL_HEALTH_CHECKER =
55              ch -> ch.eventLoop().newSucceededFuture(HttpSession.get(ch).isActive());
56  
57      final ConcurrentMap<EventLoop, KeyedChannelPool<PoolKey>> map = new ConcurrentHashMap<>();
58  
59      private final Bootstrap baseBootstrap;
60      private final SessionOptions options;
61  
62      HttpClientDelegate(Bootstrap baseBootstrap, SessionOptions options) {
63          this.baseBootstrap = requireNonNull(baseBootstrap, "baseBootstrap");
64          this.options = requireNonNull(options, "options");
65  
66          assert baseBootstrap.config().group() == null;
67      }
68  
69      @Override
70      public HttpResponse execute(ClientRequestContext ctx, HttpRequest req) throws Exception {
71          final Endpoint endpoint = ctx.endpoint().resolve().withDefaultPort(ctx.sessionProtocol().defaultPort());
72          autoFillHeaders(ctx, endpoint, req);
73  
74          final PoolKey poolKey = new PoolKey(
75                  InetSocketAddress.createUnresolved(endpoint.host(), endpoint.port()),
76                  ctx.sessionProtocol());
77  
78          final EventLoop eventLoop = ctx.eventLoop();
79          final Future<Channel> channelFuture = pool(eventLoop).acquire(poolKey);
80          final DecodedHttpResponse res = new DecodedHttpResponse(eventLoop);
81  
82          if (channelFuture.isDone()) {
83              if (channelFuture.isSuccess()) {
84                  Channel ch = channelFuture.getNow();
85                  invoke0(ch, ctx, req, res, poolKey);
86              } else {
87                  res.close(channelFuture.cause());
88              }
89          } else {
90              channelFuture.addListener((Future<Channel> future) -> {
91                  if (future.isSuccess()) {
92                      Channel ch = future.getNow();
93                      invoke0(ch, ctx, req, res, poolKey);
94                  } else {
95                      res.close(channelFuture.cause());
96                  }
97              });
98          }
99  
100         return res;
101     }
102 
103     private static void autoFillHeaders(ClientRequestContext ctx, Endpoint endpoint, HttpRequest req) {
104         requireNonNull(req, "req");
105         final HttpHeaders headers = req.headers();
106         headers.set(HttpHeaderNames.USER_AGENT, HttpHeaderUtil.USER_AGENT.toString());
107 
108         if (headers.authority() == null) {
109             final String hostname = endpoint.host();
110             final int port = endpoint.port();
111 
112             final String authority;
113             if (port == ctx.sessionProtocol().defaultPort()) {
114                 authority = hostname;
115             } else {
116                 final StringBuilder buf = new StringBuilder(hostname.length() + 6);
117                 buf.append(hostname);
118                 buf.append(':');
119                 buf.append(port);
120                 authority = buf.toString();
121             }
122 
123             headers.authority(authority);
124         }
125 
126         // Add the handlers specified in ClientOptions.
127         if (ctx.hasAttr(ClientRequestContext.HTTP_HEADERS)) {
128             headers.setAll(ctx.attr(ClientRequestContext.HTTP_HEADERS).get());
129         }
130     }
131 
132     private KeyedChannelPool<PoolKey> pool(EventLoop eventLoop) {
133         KeyedChannelPool<PoolKey> pool = map.get(eventLoop);
134         if (pool != null) {
135             return pool;
136         }
137 
138         return map.computeIfAbsent(eventLoop, e -> {
139             final Bootstrap bootstrap = baseBootstrap.clone();
140             bootstrap.group(eventLoop);
141 
142             Function<PoolKey, Future<Channel>> factory = new HttpSessionChannelFactory(bootstrap, options);
143 
144             final KeyedChannelPoolHandler<PoolKey> handler =
145                     options.poolHandlerDecorator().apply(NOOP_POOL_HANDLER);
146 
147             //TODO(inch772) handle options.maxConcurrency();
148             final KeyedChannelPool<PoolKey> newPool = new DefaultKeyedChannelPool<>(
149                     eventLoop, factory, POOL_HEALTH_CHECKER, handler, true);
150 
151             eventLoop.terminationFuture().addListener((FutureListener<Object>) f -> {
152                 map.remove(eventLoop);
153                 newPool.close();
154             });
155 
156             return newPool;
157         });
158     }
159 
160     void invoke0(Channel channel, ClientRequestContext ctx,
161                  HttpRequest req, DecodedHttpResponse res, PoolKey poolKey) {
162 
163         final HttpSession session = HttpSession.get(channel);
164         res.init(session.inboundTrafficController());
165         final SessionProtocol sessionProtocol = session.protocol();
166         if (sessionProtocol == null) {
167             res.close(ClosedSessionException.get());
168             return;
169         }
170 
171         if (session.invoke(ctx, req, res)) {
172             // Return the channel to the pool.
173             final KeyedChannelPool<PoolKey> pool = KeyedChannelPool.findPool(channel);
174             if (sessionProtocol.isMultiplex()) {
175                 pool.release(poolKey, channel);
176             } else {
177                 req.closeFuture()
178                    .handle((ret, cause) -> pool.release(poolKey, channel))
179                    .exceptionally(CompletionActions::log);
180             }
181         }
182     }
183 
184     void close() {
185         map.values().forEach(KeyedChannelPool::close);
186     }
187 }

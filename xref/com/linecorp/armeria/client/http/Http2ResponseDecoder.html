42  final class Http2ResponseDecoder extends HttpResponseDecoder implements Http2Connection.Listener,
43                                                                          Http2FrameListener {
44  
45      Http2ResponseDecoder(Channel channel) {
46          super(channel);
47      }
48

50      public void onStreamAdded(Http2Stream stream) {}
51  
52      @Override
53      public void onStreamActive(Http2Stream stream) {}
54  
55      @Override
56      public void onStreamHalfClosed(Http2Stream stream) {}
57  
58      @Override
59      public void onStreamClosed(Http2Stream stream) {}
60  
61      @Override
62      public void onStreamRemoved(Http2Stream stream) {}
63  
64      @Override
65      public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {}
66  
67      @Override
68      public void onPriorityTreeParentChanging(Http2Stream stream, Http2Stream newParent) {}
69  
70      @Override
71      public void onWeightChanged(Http2Stream stream, short oldWeight) {}
72  
73      @Override
74      public void onGoAwaySent(int lastStreamId, long errorCode, ByteBuf debugData) {}
75  
76      @Override
77      public void onGoAwayReceived(int lastStreamId, long errorCode, ByteBuf debugData) {}
78  
79      @Override
80      public void onSettingsRead(ChannelHandlerContext ctx, Http2Settings settings) {
81          ctx.fireChannelRead(settings);
82      }
83  
84      @Override
85      public void onSettingsAckRead(ChannelHandlerContext ctx) {}
86  
87      @Override
88      public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
89                                boolean endOfStream) throws Http2Exception {
90          HttpResponseWrapper res = getResponse(id(streamId), endOfStream);
91          if (res == null) {
92              throw connectionError(PROTOCOL_ERROR, "received a HEADERS frame for an unknown stream: %d",
93                                    streamId);
94          }
95  
96          final HttpHeaders converted = ArmeriaHttpUtil.toArmeria(headers);
97          if (converted.status().codeClass() != HttpStatusClass.INFORMATIONAL) {
98              res.scheduleTimeout(ctx);
99          }
100 
101         try {
102             res.write(converted);
103         } catch (Throwable t) {
104             res.close(t);
105             throw connectionError(INTERNAL_ERROR, t, "failed to consume a HEADERS frame");
106         }
107 
108         if (endOfStream) {
109             res.close();
110         }
111     }
112 
113     @Override
114     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int streamDependency,
115                               short weight, boolean exclusive, int padding, boolean endOfStream)
116             throws Http2Exception {
117         onHeadersRead(ctx, streamId, headers, padding, endOfStream);
118     }
119 
120     @Override
121     public int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding, boolean endOfStream)
122             throws Http2Exception {
123 
124         final HttpResponseWrapper res = getResponse(id(streamId), endOfStream);
125         if (res == null) {
126             throw connectionError(PROTOCOL_ERROR, "received a DATA frame for an unknown stream: %d",
127                                   streamId);
128         }
129 
130         final int dataLength = data.readableBytes();
131         final long maxContentLength = res.maxContentLength();
132         if (maxContentLength > 0 && res.writtenBytes() > maxContentLength - dataLength) {
133             res.close(ContentTooLargeException.get());
134             throw connectionError(INTERNAL_ERROR,
135                                   "content length too large: %d + %d > %d (stream: %d)",
136                                   res.writtenBytes(), dataLength, maxContentLength, streamId);
137         }
138 
139         try {
140             res.write(HttpData.of(data));
141         } catch (Throwable t) {
142             res.close(t);
143             throw connectionError(INTERNAL_ERROR, t, "failed to consume a DATA frame");
144         }
145 
146         if (endOfStream) {
147             res.close();
148         }
149 
150         // All bytes have been processed.
151         return dataLength + padding;
152     }
153 
154     @Override
155     public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception {
156         final HttpResponseWriter res = removeResponse(id(streamId));
157         if (res == null) {
158             throw connectionError(PROTOCOL_ERROR,
159                                   "received a RST_STREAM frame for an unknown stream: %d", streamId);
160         }
161 
162         res.close(ClosedSessionException.get());
163     }
164 
165     @Override
166     public void onPushPromiseRead(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
167                                   Http2Headers headers, int padding) {}
168 
169     @Override
170     public void onPriorityRead(ChannelHandlerContext ctx, int streamId, int streamDependency, short weight,
171                                boolean exclusive) {}
172 
173     @Override
174     public void onPingRead(ChannelHandlerContext ctx, ByteBuf data) {}
175 
176     @Override
177     public void onPingAckRead(ChannelHandlerContext ctx, ByteBuf data) {}
178 
179     @Override
180     public void onGoAwayRead(ChannelHandlerContext ctx, int lastStreamId, long errorCode, ByteBuf debugData) {}
181 
182     @Override
183     public void onWindowUpdateRead(ChannelHandlerContext ctx, int streamId, int windowSizeIncrement) {}
184 
185     @Override
186     public void onUnknownFrame(ChannelHandlerContext ctx, byte frameType, int streamId, Http2Flags flags,
187                                ByteBuf payload) {}
188 
189     private static int id(int streamId) {
190         return streamId - 1 >>> 1;
191     }
192 }

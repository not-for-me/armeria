122     public static boolean isContentAlwaysEmpty(HttpStatus status) {
123         if (status.codeClass() == HttpStatusClass.INFORMATIONAL) {
124             return true;
125         }
126 
127         switch (status.code()) {
128             case 204: case 205: case 304:
129                 return true;
130         }
131 
132         return false;
133     }

135     public static HttpHeaders toArmeria(Http2Headers headers) {
136         final HttpHeaders converted = new DefaultHttpHeaders(false, headers.size());
137         for (Entry<CharSequence, CharSequence> e : headers) {
138             converted.add(AsciiString.of(e.getKey()), e.getValue().toString());
139         }
140         return converted;
141     }

143     /**
144      * Converts the given HTTP/1.x headers into HTTP/2 headers.
145      * The following headers are only used if they can not be found in the {@code HOST} header or the
146      * {@code Request-Line} as defined by <a href="https://tools.ietf.org/html/rfc7230">rfc7230</a>
147      * <ul>
148      * <li>{@link ExtensionHeaderNames#SCHEME}</li>
149      * </ul>
150      * {@link ExtensionHeaderNames#PATH} is ignored and instead extracted from the {@code Request-Line}.
151      */
152     public static HttpHeaders toArmeria(HttpMessage in) {
153         io.netty.handler.codec.http.HttpHeaders inHeaders = in.headers();
154         final HttpHeaders out = new DefaultHttpHeaders(true, inHeaders.size());
155         if (in instanceof HttpRequest) {
156             HttpRequest request = (HttpRequest) in;
157             URI requestTargetUri = URI.create(request.uri());
158             out.path(toHttp2Path(requestTargetUri));
159             out.method(HttpMethod.valueOf(request.method().name()));
160             setHttp2Scheme(inHeaders, requestTargetUri, out);
161 
162             if (!isOriginForm(requestTargetUri) && !isAsteriskForm(requestTargetUri)) {
163                 // Attempt to take from HOST header before taking from the request-line
164                 String host = inHeaders.getAsString(HttpHeaderNames.HOST);
165                 setHttp2Authority(host == null || host.isEmpty() ? requestTargetUri.getAuthority() : host, out);
166             }
167         } else if (in instanceof HttpResponse) {
168             HttpResponse response = (HttpResponse) in;
169             out.status(response.status().code());
170         }
171 
172         // Add the HTTP headers which have not been consumed above
173         toArmeria(inHeaders, out);
174         return out;
175     }

177     public static HttpHeaders toArmeria(io.netty.handler.codec.http.HttpHeaders inHeaders) {
178         if (inHeaders.isEmpty()) {
179             return HttpHeaders.EMPTY_HEADERS;
180         }
181 
182         final HttpHeaders out = new DefaultHttpHeaders(true, inHeaders.size());
183         toArmeria(inHeaders, out);
184         return out;
185     }

187     public static void toArmeria(io.netty.handler.codec.http.HttpHeaders inHeaders, HttpHeaders out) {
188         final Iterator<Entry<CharSequence, CharSequence>> i = inHeaders.iteratorCharSequence();
189         while (i.hasNext()) {
190             final Entry<CharSequence, CharSequence> entry = i.next();
191             final AsciiString aName = AsciiString.of(entry.getKey()).toLowerCase();
192             if (!HTTP_TO_HTTP2_HEADER_BLACKLIST.contains(aName)) {
193                 // https://tools.ietf.org/html/rfc7540#section-8.1.2.2 makes a special exception for TE
194                 if (aName.contentEqualsIgnoreCase(HttpHeaderNames.TE) &&
195                     !AsciiString.contentEqualsIgnoreCase(entry.getValue(), HttpHeaderValues.TRAILERS)) {
196                     continue;
197                 }
198 
199                 out.add(aName, entry.getValue().toString());
200             }
201         }
202     }

204     /**
205      * Generate a HTTP/2 {code :path} from a URI in accordance with
206      * <a href="https://tools.ietf.org/html/rfc7230#section-5.3">rfc7230, 5.3</a>.
207      */
208     private static String toHttp2Path(URI uri) {
209         StringBuilder pathBuilder = new StringBuilder(length(uri.getRawPath()) +
210                                                       length(uri.getRawQuery()) + length(uri.getRawFragment()) + 2);
211         if (!isNullOrEmpty(uri.getRawPath())) {
212             pathBuilder.append(uri.getRawPath());
213         }
214         if (!isNullOrEmpty(uri.getRawQuery())) {
215             pathBuilder.append('?');
216             pathBuilder.append(uri.getRawQuery());
217         }
218         if (!isNullOrEmpty(uri.getRawFragment())) {
219             pathBuilder.append('#');
220             pathBuilder.append(uri.getRawFragment());
221         }
222 
223         return pathBuilder.toString();
224     }

226     private static void setHttp2Authority(String authority, HttpHeaders out) {
227         // The authority MUST NOT include the deprecated "userinfo" subcomponent
228         if (authority != null) {
229             int endOfUserInfo = authority.indexOf('@');
230             if (endOfUserInfo < 0) {
231                 out.authority(authority);
232             } else if (endOfUserInfo + 1 < authority.length()) {
233                 out.authority(authority.substring(endOfUserInfo + 1));
234             } else {
235                 throw new IllegalArgumentException("authority: " + authority);
236             }
237         }
238     }

240     private static void setHttp2Scheme(io.netty.handler.codec.http.HttpHeaders in, URI uri, HttpHeaders out) {
241         String value = uri.getScheme();
242         if (value != null) {
243             out.scheme(value);
244             return;
245         }
246 
247         // Consume the Scheme extension header if present
248         CharSequence cValue = in.get(ExtensionHeaderNames.SCHEME.text());
249         if (cValue != null) {
250             out.scheme(cValue.toString());
251         } else {
252             out.scheme("unknown");
253         }
254     }

256     public static Http2Headers toNettyHttp2(HttpHeaders inputHeaders) {
257         final Http2Headers outputHeaders = new DefaultHttp2Headers(false, inputHeaders.size());
258         outputHeaders.set(inputHeaders);
259         outputHeaders.remove(HttpHeaderNames.TRANSFER_ENCODING);
260         outputHeaders.remove(HttpHeaderNames.TRAILER);
261         return outputHeaders;
262     }

264     public static io.netty.handler.codec.http.HttpHeaders toNettyHttp1(HttpHeaders inputHeaders) {
265         final io.netty.handler.codec.http.DefaultHttpHeaders outputHeaders =
266                 new io.netty.handler.codec.http.DefaultHttpHeaders();
267         for (Entry<AsciiString, String> e : inputHeaders) {
268             final AsciiString name = e.getKey();
269             if (name.isEmpty() || HTTP2_TO_HTTP_HEADER_BLACKLIST.contains(name)) {
270                 continue;
271             }
272             outputHeaders.add(name, e.getValue());
273         }
274         return outputHeaders;
275     }

277     /**
278      * Translate and add HTTP/2 headers to HTTP/1.x headers.
279      *
280      * @param streamId The stream associated with {@code sourceHeaders}.
281      * @param inputHeaders The HTTP/2 headers to convert.
282      * @param outputHeaders The object which will contain the resulting HTTP/1.x headers..
283      * @param httpVersion What HTTP/1.x version {@code outputHeaders} should be treated as when doing the conversion.
284      * @param isTrailer {@code true} if {@code outputHeaders} should be treated as trailing headers.
285      *                  {@code false} otherwise.
286      * @param isRequest {@code true} if the {@code outputHeaders} will be used in a request message.
287      *                  {@code false} for response message.
288      *
289      * @throws Http2Exception If not all HTTP/2 headers can be translated to HTTP/1.x.
290      */
291     public static void toNettyHttp1(
292             int streamId, HttpHeaders inputHeaders, io.netty.handler.codec.http.HttpHeaders outputHeaders,
293             HttpVersion httpVersion, boolean isTrailer, boolean isRequest) throws Http2Exception {
294 
295         final Http2ToHttpHeaderTranslator translator =
296                 new Http2ToHttpHeaderTranslator(outputHeaders, isRequest);
297         try {
298             for (Entry<AsciiString, String> entry : inputHeaders) {
299                 translator.translate(entry);
300             }
301         } catch (Throwable t) {
302             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
303         }
304 
305         outputHeaders.remove(HttpHeaderNames.TRANSFER_ENCODING);
306         outputHeaders.remove(HttpHeaderNames.TRAILER);
307         if (!isTrailer) {
308             HttpUtil.setKeepAlive(outputHeaders, httpVersion, true);
309         }
310     }

312     /**
313      * Utility which translates HTTP/2 headers to HTTP/1 headers.
314      */
315     private static final class Http2ToHttpHeaderTranslator {
316         /**
317          * Translations from HTTP/2 header name to the HTTP/1.x equivalent.
318          */
319         private static final CharSequenceMap REQUEST_HEADER_TRANSLATIONS = new CharSequenceMap();
320         private static final CharSequenceMap RESPONSE_HEADER_TRANSLATIONS = new CharSequenceMap();
321         static {
322             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.AUTHORITY.value(),
323                                              HttpHeaderNames.HOST);
324             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.SCHEME.value(),
325                                              ExtensionHeaderNames.SCHEME.text());
326             REQUEST_HEADER_TRANSLATIONS.add(RESPONSE_HEADER_TRANSLATIONS);
327             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.PATH.value(),
328                                              ExtensionHeaderNames.PATH.text());
329         }
330 
331         private final io.netty.handler.codec.http.HttpHeaders output;
332         private final CharSequenceMap translations;
333 
334         /**
335          * Create a new instance
336          *
337          * @param output The HTTP/1.x headers object to store the results of the translation
338          * @param request if {@code true}, translates headers using the request translation map. Otherwise uses the
339          *        response translation map.
340          */
341         Http2ToHttpHeaderTranslator(io.netty.handler.codec.http.HttpHeaders output,
342                                     boolean request) {
343 
344             this.output = output;
345             translations = request ? REQUEST_HEADER_TRANSLATIONS : RESPONSE_HEADER_TRANSLATIONS;
346         }
347 
348         public void translate(Entry<AsciiString, String> entry) {
349             final AsciiString name = entry.getKey();
350             final String value = entry.getValue();
351             AsciiString translatedName = translations.get(name);
352             if (translatedName != null) {
353                 output.add(translatedName, value);
354                 return;
355             }
356 
357             // https://tools.ietf.org/html/rfc7540#section-8.1.2.3
358             if (name.isEmpty() || HTTP2_TO_HTTP_HEADER_BLACKLIST.contains(name)) {
359                 return;
360             }
361 
362             if (HttpHeaderNames.COOKIE.equals(name)) {
363                 // combine the cookie values into 1 header entry.
364                 // https://tools.ietf.org/html/rfc7540#section-8.1.2.5
365                 String existingCookie = output.get(HttpHeaderNames.COOKIE);
366                 output.set(HttpHeaderNames.COOKIE,
367                            existingCookie != null ? existingCookie + "; " + value : value);
368             } else {
369                 output.add(name, value);
370             }
371         }
372     }

17  package com.linecorp.armeria.common.logging;
18  
19  import java.util.Objects;
20  
21  import com.linecorp.armeria.common.RequestContext;
22  
23  public interface MessageLogConsumer {
24  
25      /**
26       * Invoked when a request has been streamed.
27       */
28      void onRequest(RequestContext ctx, RequestLog req) throws Exception;
29  
30      /**
31       * Invoked when a response has been streamed.
32       */
33      void onResponse(RequestContext ctx, ResponseLog res) throws Exception;
34  
35      default MessageLogConsumer andThen(MessageLogConsumer other) {
36          Objects.requireNonNull(other, "other");
37  
38          final MessageLogConsumer first = this;
39          final MessageLogConsumer second = other;
40  
41          return new MessageLogConsumer() {
42              @Override
43              public void onRequest(RequestContext ctx, RequestLog req) throws Exception {
44                  MessageLogConsumerInvoker.invokeOnRequest(first, ctx, req);
45                  second.onRequest(ctx, req);
46              }
47  
48              @Override
49              public void onResponse(RequestContext ctx, ResponseLog res) throws Exception {
50                  MessageLogConsumerInvoker.invokeOnResponse(first, ctx, res);
51                  second.onResponse(ctx, res);
52              }
53          };
54      }
55  }

17  package com.linecorp.armeria.common;
18  
19  import static com.google.common.net.MediaType.create;
20  
21  import java.util.Collections;
22  import java.util.EnumSet;
23  import java.util.HashMap;
24  import java.util.Locale;
25  import java.util.Map;
26  import java.util.Optional;
27  import java.util.Set;
28  
29  import javax.annotation.Nullable;
30  
31  import com.google.common.net.MediaType;
32  
33  /**
34   * Serialization format of a remote procedure call and its reply.
35   */
36  public enum SerializationFormat {
37  
38      /**
39       * No serialization format. Used when no serialization/deserialization is desired.
40       */
41      NONE("none", create("application", "x-none")),
42  
43      /**
44       * Unknown serialization format. Used when some serialization format is desired but the server
45       * failed to understand/recognize it.
46       */
47      UNKNOWN("unknown", create("application", "x-unknown")),
48  
49      /**
50       * Thrift TBinary serialization format
51       */
52      THRIFT_BINARY("tbinary", create("application", "x-thrift").withParameter("protocol", "TBINARY")),
53  
54      /**
55       * Thrift TCompact serialization format
56       */
57      THRIFT_COMPACT("tcompact", create("application", "x-thrift").withParameter("protocol", "TCOMPACT")),
58  
59      /**
60       * Thrift TJSON serialization format
61       */
62      THRIFT_JSON("tjson", create("application", "x-thrift").withParameter("protocol", "TJSON")),
63  
64      /**
65       * Thrift TText serialization format. This format is not optimized for performance or backwards
66       * compatibility and should only be used in non-production use cases like debugging.
67       */
68      THRIFT_TEXT("ttext", create("application", "x-thrift").withParameter("protocol", "TTEXT"));
69  
70      private static final Set<SerializationFormat> THRIFT_FORMATS = Collections.unmodifiableSet(
71              EnumSet.of(THRIFT_BINARY, THRIFT_COMPACT, THRIFT_JSON, THRIFT_TEXT));
72  
73      private static final Map<String, Optional<SerializationFormat>> PROTOCOL_TO_THRIFT_FORMATS;
74  
75      static {
76          Map<String, Optional<SerializationFormat>> protocolToThriftFormats = new HashMap<>();
77          for (SerializationFormat f : THRIFT_FORMATS) {
78              protocolToThriftFormats.put(f.uriText(), Optional.of(f));
79          }
80          PROTOCOL_TO_THRIFT_FORMATS = Collections.unmodifiableMap(protocolToThriftFormats);
81      }
82  
83      /**
84       * Returns the set of all known Thrift serialization formats. This method is useful when determining if a
85       * {@link SerializationFormat} is Thrift or not.
86       * e.g. {@code if (SerializationFormat.ofThrift().contains(serFmt)) { ... }}
87       */
88      public static Set<SerializationFormat> ofThrift() {
89          return THRIFT_FORMATS;
90      }
91  
92      /**
93       * Returns the serialization format corresponding to the passed in {@code mediaType}, or
94       * {@link Optional#empty} if the media type is not recognized. {@code null} is treated as an unknown
95       * mimetype.
96       */
97      public static Optional<SerializationFormat> fromMediaType(@Nullable String mediaType) {
98          if (mediaType == null || mediaType.isEmpty()) {
99              return Optional.empty();
100         }
101 
102         final int semicolonIdx = mediaType.indexOf(';');
103         final String paramPart;
104         if (semicolonIdx >= 0) {
105             paramPart = mediaType.substring(semicolonIdx).toLowerCase(Locale.US);
106             mediaType = mediaType.substring(0, semicolonIdx).toLowerCase(Locale.US).trim();
107         } else {
108             paramPart = null;
109             mediaType = mediaType.toLowerCase(Locale.US).trim();
110         }
111 
112         if ("application/x-thrift".equals(mediaType)) {
113             return fromThriftMediaType(paramPart);
114         }
115 
116         if (NONE.mediaType().toString().equals(mediaType)) {
117             return Optional.of(NONE);
118         }
119 
120         return Optional.empty();
121     }
122 
123     private static Optional<SerializationFormat> fromThriftMediaType(String params) {
124         final String protocol = MimeTypeParams.find(params, "protocol");
125         return PROTOCOL_TO_THRIFT_FORMATS.getOrDefault(protocol, Optional.empty());
126     }
127 
128     private final String uriText;
129     private final MediaType mediaType;
130 
131     SerializationFormat(String uriText, MediaType mediaType) {
132         this.uriText = uriText;
133         this.mediaType = mediaType;
134     }
135 
136     /**
137      * Returns the textual representation of this format for use in a {@link Scheme}.
138      */
139     public String uriText() {
140         return uriText;
141     }
142 
143     public MediaType mediaType() {
144         return mediaType;
145     }
146 }

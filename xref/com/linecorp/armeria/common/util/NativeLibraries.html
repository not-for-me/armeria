17  package com.linecorp.armeria.common.util;
18  
19  import java.util.concurrent.atomic.AtomicBoolean;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import io.netty.channel.epoll.Epoll;
25  import io.netty.handler.ssl.OpenSsl;
26  
27  /**
28   * Reports the availability of the native libraries used by Armeria
29   */
30  public final class NativeLibraries {
31  
32      private static final Logger logger = LoggerFactory.getLogger(NativeLibraries.class);
33      private static final AtomicBoolean reported = new AtomicBoolean();
34  
35      private static final boolean USE_EPOLL =
36              !"false".equals(System.getProperty("com.linecorp.armeria.useEpoll", "true"));
37  
38      private static final boolean USE_OPENSSL =
39              !"false".equals(System.getProperty("com.linecorp.armeria.useOpenSsl", "true"));
40  
41      /**
42       * Logs the availability of the native libraries used by Armeria. This method does nothing if it was
43       * called once before.
44       */
45      public static void report() {
46          if (!reported.compareAndSet(false, true)) {
47              return;
48          }
49  
50          if (USE_EPOLL) {
51              logger.info("/dev/epoll: " +
52                          (Epoll.isAvailable() ? "yes"
53                                               : "no (" + filterCause(Epoll.unavailabilityCause()) + ')'));
54          } else {
55              logger.info("/dev/epoll: disabled");
56          }
57  
58          if (USE_OPENSSL) {
59              logger.info("OpenSSL: " +
60                          (OpenSsl.isAvailable() ? "yes (" + OpenSsl.versionString() + ", " +
61                                                             OpenSsl.version() + ')'
62                                                 : "no (" + filterCause(OpenSsl.unavailabilityCause()) + ')'));
63          } else {
64              logger.info("OpenSSL: disabled");
65          }
66      }
67  
68      private static Throwable filterCause(Throwable cause) {
69          if (cause instanceof ExceptionInInitializerError) {
70              return cause.getCause();
71          }
72  
73          return cause;
74      }
75  
76      /**
77       * Returns {@code true} if JNI-based {@code /dev/epoll} transport is available.
78       */
79      public static boolean isEpollAvailable() {
80          report();
81          return USE_EPOLL && Epoll.isAvailable();
82      }
83  
84      /**
85       * Returns {@code true} if JNI-based OpenSSL/BoringSSL/LibreSSL transport is available.
86       */
87      public static boolean isOpenSslAvailable() {
88          report();
89          return USE_OPENSSL && OpenSsl.isAvailable();
90      }
91  
92      private NativeLibraries() {}
93  }
